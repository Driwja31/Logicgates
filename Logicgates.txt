#Code to implement logic gate using python # Define the logic gate functionsdef AND_gate(a, b):    return a and bdef OR_gate(a, b):    return a or bdef NOT_gate(a):    return not adef NAND_gate(a, b):    return not (a and b)def NOR_gate(a, b):    return not (a or b)def XOR_gate(a, b):    return (a and not b) or (not a and b)# Prompt the user to choose a gategate_choice = int(input("Choose a gate to test: \n1. AND gate\n2. OR gate\n3. NOT gate for a\n4. NOT gate for b\n5. NAND gate\n6. NOR gate\n7. XOR gate\n"))# Prompt the user for input valuesa = int(input("Enter the value of input a (0 or 1): "))b = int(input("Enter the value of input b (0 or 1): "))# Calculate the output of the selected gateif gate_choice == 1:    output = AND_gate(a, b)    gate_name = "AND gate"elif gate_choice == 2:    output = OR_gate(a, b)    gate_name = "OR gate"elif gate_choice == 3:    output = NOT_gate(a)    gate_name = "NOT gate for a"elif gate_choice == 4:    output = NOT_gate(b)    gate_name = "NOT gate for b"elif gate_choice == 5:    output = NAND_gate(a, b)    gate_name = "NAND gate"elif gate_choice == 6:    output = NOR_gate(a, b)    gate_name = "NOR gate"elif gate_choice == 7:    output = XOR_gate(a, b)    gate_name = "XOR gate"else:    print("Invalid choice, please try again.")    exit()# Print the name and output of the selected gateprint("Output of the", gate_name, "is", output)
